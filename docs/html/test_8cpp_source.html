114 TestableGraph getTranspose() {
115 TestableGraph gT;
116 gT.setVert(vertices);
117 gT.setDir(directed);
118 if (matrix) {
119 gT.matrix = true;
120 gT.graph.assign(vertices, vector<int>(vertices, 0));
121 for (int i = 0; i < vertices; i++)
122 for (int j = 0; j < vertices; j++)
123 if (graph[i][j])
124 gT.graph[j][i] = 1;
125 }
126 else {
127 gT.matrix = false;
128 gT.graph.resize(vertices);
129 for (int i = 0; i < vertices; i++)
130 for (int u : graph[i])
131 gT.graph[u].push_back(i);
132 }
133 return gT;
134 }
