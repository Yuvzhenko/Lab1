472void Graph::distanceBetween() {

474 int start, end;

478
479 vector<bool> visited(vertices, false);

481 vector<int> queue;
482
483 int front = 0;
484
485 queue.push_back(start);
486 visited[start] = true;
487 dist[start] = 0;
488
489 while (front < queue.size()) {
490 int u = queue[front++];
491 if (matrix) {
492 for (int v = 0; v < vertices; v++) {
493 if (graph[u][v] && !visited[v]) {
494 visited[v] = true;
495 dist[v] = dist[u] + 1;
496 queue.push_back(v);

498 }
499 }

501 for (int v : graph[u]) {

503 visited[v] = true;

507 }
